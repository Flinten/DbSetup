025    package com.ninja_squad.dbsetup.destination;
026    
027    import java.sql.Connection;
028    import java.sql.SQLException;
029    
030    import javax.annotation.Nonnull;
031    import javax.annotation.concurrent.Immutable;
032    import javax.sql.DataSource;
033    
034    import com.ninja_squad.dbsetup.util.Preconditions;
035    
036    /**
037     * A destination which wraps a DataSource and gets its connection from the wrapped DataSource
038     * @author JB Nizet
039     */
040    @Immutable
041    public final class DataSourceDestination implements Destination {
042        private final DataSource dataSource;
043    
044        /**
045         * Constructor
046         * @param dataSource the wrapped DataSource
047         */
048        public DataSourceDestination(@Nonnull DataSource dataSource) {
049            Preconditions.checkNotNull(dataSource, "dataSource may not be null");
050            this.dataSource = dataSource;
051        }
052    
053        @Override
054        public Connection getConnection() throws SQLException {
055            return dataSource.getConnection();
056        }
057    
058        @Override
059        public String toString() {
060            return "DataSourceDestination [dataSource=" + dataSource + "]";
061        }
062    
063        @Override
064        public int hashCode() {
065            return dataSource.hashCode();
066        }
067    
068        @Override
069        public boolean equals(Object obj) {
070            if (this == obj) {
071                return true;
072            }
073            if (obj == null) {
074                return false;
075            }
076            if (getClass() != obj.getClass()) {
077                return false;
078            }
079            DataSourceDestination other = (DataSourceDestination) obj;
080            return dataSource.equals(other.dataSource);
081        }
082    }
