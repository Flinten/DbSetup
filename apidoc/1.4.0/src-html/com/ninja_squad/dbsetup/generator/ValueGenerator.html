025    package com.ninja_squad.dbsetup.generator;
026    
027    /**
028     * A value generator allows generating values for a specific column in a sequence of inserts. This is useful when you
029     * don't want to specify a value for each of the inserted rows in a table, and when a default value is not an option
030     * either because, for example, the column has a unique constraint.
031     * @param <T> the type of value that this generator generates
032     *
033     * @see ValueGenerators for useful implementations of this interface
034     * @author JB Nizet
035     */
036    public interface ValueGenerator<T> {
037        /**
038         * Called each time a new row is inserted, to get the value to insert in the column using this value generator.
039         * @return the value to insert in the column associated with this generator.
040         */
041        T nextValue();
042    }
