025    package com.ninja_squad.dbsetup.bind;
026    
027    import java.sql.PreparedStatement;
028    import java.sql.SQLException;
029    
030    import com.ninja_squad.dbsetup.DbSetupTracker;
031    
032    /**
033     * An object which binds a value to a prepared statement parameter. It's advised to make implementations of this
034     * interface immutable, and to make them implement equals and hashCode in order for {@link DbSetupTracker} to function
035     * properly, or to make them singletons.
036     * @author JB Nizet
037     */
038    public interface Binder {
039        /**
040         * Binds the given value to the given parameter in the given prepared statement.
041         * @param statement the statement to bind the parameter to
042         * @param param The index of the parameter to bind in the statement
043         * @param value The value to bind (may be <code>null</code>)
044         * @throws SQLException if the binding throws a {@link SQLException}
045         */
046        void bind(PreparedStatement statement, int param, Object value) throws SQLException;
047    }
